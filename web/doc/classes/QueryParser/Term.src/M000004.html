  def initialize(data)
    @type = 'term'
    @data = data
    @was_reduced = false

    if @data == nil then
      @data = ''
    else
      case @data.downcase
        when "("
          @type = "open"
        when ")"
          @type = "close"
        when "and", "or", "not"
          @type = 'op'
          @data = @data.downcase
      end
    end
  end
