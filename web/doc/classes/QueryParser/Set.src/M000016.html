  def reduce
    r = Array.new
    @was_reduced = false

    @data.each do |i|
      if self.class == i.class then
        @was_reduced = true
        i.contents.each do |c|
          r << c.reduce
        end
      else
        x = i.reduce
        if x.reduced? then
          @was_reduced = true
        end
        r << x
      end
    end

    if r.size == 1 then
      @was_reduced = true
      r[0].set_reduced
      return r.first
    else
      @data = r
      return self
    end
  end
