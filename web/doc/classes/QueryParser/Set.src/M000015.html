  def lucene(field, similarity = nil)
    r = Array.new
    @data.each do |i|
      x = ''
      if self.class == QueryParser::And and i.class != QueryParser::Not then
        x = '+'
      end
      x << i.lucene(field, similarity)
      r << x
    end

    if r.size == 1 then
      "#{r[0]}"
    else
      "(#{r.join(' ')})"
    end
  end
